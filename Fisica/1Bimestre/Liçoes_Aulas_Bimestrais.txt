    function acessarTrilha() {
        const sel = document.getElementById('selAluno');
        if(!sel.value) return alert("⚠️ Selecione seu nome!");
        estado.aluno = sel.value;
        estado.inicio = new Date();
        document.getElementById('tela-login').style.display = 'none';

    function renderizarQuestao() {
        const q = QUESTOES[estado.index];
        document.getElementById('barra').style.width = ((estado.index / QUESTOES.length) * 100) + "%";

            <div class="d-grid gap-2">
                ${q.r.map((opt, i) => `

                `).join('')}
            </div>
        `;

    function validar(idx) {
        const q = QUESTOES[estado.index];
        const escolhida = q.r[idx];
        
        estado.mapa.push(escolhida.diag);
        if(escolhida.c) estado.acertos++;
        
        estado.index++;
        if(estado.index < QUESTOES.length) renderizarQuestao();
        else finalizar();
    }

    function finalizar() {
        document.getElementById('barra').style.width = "100%";
        document.getElementById('titulo-sessao').innerText = "CONCLUÍDO";
        document.getElementById('conteudo-dinamico').innerHTML = `
            <div class="text-center py-4">
                <h3 class="text-success fw-bold">Trilha Concluída!</h3>
                <p class="fs-4">Acertos: ${estado.acertos} de ${QUESTOES.length}</p>
                <button id="btn-enviar" class="btn btn-success btn-lg px-5 shadow" onclick="enviarRelatorio()">🚀 ENVIAR PARA O PROFESSOR</button>
            </div>
        `;
    }
